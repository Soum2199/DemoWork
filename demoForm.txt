// //  addEventListener is used to listen to the events, such as performing the click events, 
// // classlist  vs classList

const inputs = document.getElementById("pass")
const button = document.querySelector("button")

button.addEventListener("click", function(){
  if(inputs.type == "text"){
    button.innerText = "Show"
    inputs.type = "password"
  }
  else {
    button.innerText = "Hide"
    inputs.type = "text"
  }
})
