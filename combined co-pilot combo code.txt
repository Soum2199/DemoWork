const turns = document.getElementById('turns');
const boxes = Array.from(document.querySelectorAll(".box"));
const status = document.getElementById("status");
const reseti = document.getElementById('reseti');
const themeToggle = document.getElementById('themeToggle');

let currentPlayer = "X";
let arr = Array(9).fill(null);
let winnerStatus = false;
let winningLine = null;

// initial UI
updateTurnText();
status.textContent = '';

function updateTurnText() {
  turns.textContent = winnerStatus ? '' : `${currentPlayer}'s turn`;
}

function checkWin() {
  const combos = [
    [0,1,2],
    [3,4,5],
    [6,7,8],
    [0,3,6],
    [1,4,7],
    [2,5,8],
    [0,4,8],
    [2,4,6]
  ];

  for (let combo of combos) {
    const [a,b,c] = combo;
    if (arr[a] && arr[a] === arr[b] && arr[b] === arr[c]) {
      winnerStatus = true;
      winningLine = combo;
      status.textContent = `${arr[a]} is the winner!`;
      highlightWinning(combo);
      return;
    }
  }

  // Check draw
  if (arr.every(cell => cell !== null)) {
    status.textContent = "Game is a draw";
    winnerStatus = true; // disable further moves
  }
}

function highlightWinning(combo) {
  combo.forEach(i => {
    const el = document.getElementById(String(i));
    if (el) el.classList.add('win');
  });
}

function swapPlayers() {
  currentPlayer = currentPlayer === "X" ? "O" : "X";
  updateTurnText();
}

function handleBoxClick(e) {
  const idx = Number(this.id);
  if (winnerStatus) return; // stop if game over
  if (arr[idx] !== null) return; // already filled

  arr[idx] = currentPlayer;
  this.textContent = currentPlayer;

  checkWin();

  if (!winnerStatus) {
    swapPlayers();
  }
}

// keyboard accessibility: press Enter or Space to play when box focused
function handleBoxKeydown(e) {
  if (e.key === 'Enter' || e.key === ' ') {
    e.preventDefault();
    this.click();
  }
}

// attach events
boxes.forEach(box => {
  box.addEventListener('click', handleBoxClick);
  box.addEventListener('keydown', handleBoxKeydown);
});

reseti.addEventListener("click", function () {
  currentPlayer = "X";
  arr = Array(9).fill(null);
  winnerStatus = false;
  winningLine = null;
  boxes.forEach(b => {
    b.textContent = "";
    b.classList.remove('win');
  });
  status.textContent = '';
  updateTurnText();
});

// theme toggle (guarded)
if (themeToggle) {
  themeToggle.addEventListener("click", function () {
    document.body.classList.toggle("darky");
    themeToggle.classList.toggle("darky");
  });
}
