      <button type="submit" id="submits">Add task</button>
      <br />
      <ul>      </ul>
      <script>
const submits = document.getElementById("submits");
const ul = document.querySelector("ul")
const inputTask = document.getElementById('task')

submits.addEventListener("click", function () {
  let myTask = inputTask.value;
  const li = document.createElement("li")
  li.innerHTML = `${myTask} <button id="del">Delete</button>`
  console.log(myTask)
  ul.append(li)
  inputTask.value = ''
  
  // for some delete functions.
  // let del = document.getElementById('del')
    let del = li.getElementById('del')
  del.addEventListener("click", function () {
    li.remove()
  })
  
})
</script>
  </body>
</html>

  ul.append(li)

  // Correct way to select the delete button inside this li
