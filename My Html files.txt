let sec = 0, min=0, hr=0;

function showTime2() {
  // sec = 0;
  sec++;
  if(sec > 59){
    sec = 0;
    min++;
    if(min > 59){
      min = 0;
      hr++;
      if(hr > 23){
        sec = 0;
        min=0; hr=0;
        title.textContent = "00:00:00";
      }
    }
  }
  let finalSec = sec > 10 ? `0${sec}` : sec;
  let finalMin = min > 10 ? `0${min}` : min;
  let finalHr = hr > 10 ? `0${hr}` : hr;
  title.innerText = `${finalHr}:${finalMin}:${finalSec}`;
}
